# qiankun 框架应用实例

# 介绍

  • 基于 Single-Spa 封装的沙箱工具
  • 劫持子项目资源，将 window 绑定到 Proxy 对象上，防止污染全局变量，达到子项目之间的脚本隔离

# 项目结构

┌── dist # 生产包
│   ├── micro-app-main # 基座外网生产包
│   ├── micro-app-search # 子项目 A 外网生产包
│   ├── micro-app-search # 子项目 B 外网生产包
│   ├── judicial-bigdata-main 内网.zip # 基座内网压缩包
│   ├── judicial-bigdata-search 内网.zip # 子项目 A 内网压缩包
│   └── judicial-bigdata-visual 内网.zip # 子项目 B 内网压缩包
├── micro-app-main # 基座开发入口
├── micro-app-search # 子项目 A 开发入口
├── micro-app-visual # 子项目 B 开发入口
└── package.json # 批量执行所有项目的 npm 命令

// microApp.js
import {
  registerMicroApps,
  addGlobalUncaughtErrorHandler,
  start,
  setDefaultMountApp,
  initGlobalState,
} from 'qiankun'

class MicroApp {
  #option

  constructor(config = {}) {
    // 创建子项目配置
    this.#option = Object.freeze(
      Object.entries(config[process.env.NODE_ENV ?? 'development']).reduce(
        (arr, [key, val]) => {
          arr.push(
            Object.freeze({
              name: key, // 名称
              entry: val, // 入口
              container: '#micro-app', // 挂载节点
              activeRule: `/${key}`, // 触发挂载对应子项目的路径
            })
          )
          return arr
        },
        []
      )
    )
  }

  static $mount() {
    // 注册子项目及生命周期
    registerMicroApps(this.#option, {
      beforeLoad: () => Promise.resolve(),
      afterMount: () => Promise.resolve(),
    })
    // 初始化调用子项目
    setDefaultMountApp('/micro-app-search/login')
    // 监听加载子项目的报错
    addGlobalUncaughtErrorHandler((event) => {
      console.error(event)
    })
    // 执行微前端
    start()
  }
}
export { MicroApp, initGlobalState }

import Vue from 'vue'
import VueRouter from 'vue-router'
import constantRoutes from './router'
import ElementUI from './components/elementui'
// 引入封装好的microChildApp
import {
  actions, // 通信
  inject, // 注入微前端环境
  render, // 渲染
  bootstrap, // 生命周期
  mount, // 生命周期
  unmount, // 生命周期
} from './microChildApp'

Vue.prototype.$actions = actions // 全局调用微前端通信
Vue.use(VueRouter)
Vue.use(ElementUI)

// 定义初始化Vue的方法
function InitVue(router) {
  return new Vue({
    router,
    store,
    render: (h) => h(App),
  }).$mount('#app')
}

// 定义初始化VueRouter的方法
function InitVueRouter(constantRoutes) {
  return new VueRouter({
    base: window.__POWERED_BY_QIANKUN__ ? '/micro-app-search' : '/',
    mode: 'history',
    routes: constantRoutes,
  })
}

// 判断是否为微前端环境
if (!window.__POWERED_BY_QIANKUN__) {
  // 独立运行子项目时主动加载基座公共资源 _main:../micro-app-main/src
  import('_main/assets/iconfont/iconfont.css')
  import('_main/assets/iconfont/iconfont')
  import('element-ui/lib/theme-chalk/index.css')
  import('_main/styles/index.scss')
  // 渲染
  render({
    props: {},
    InitVue,
    InitVueRouter,
  })
} else {
  // 子项目资源重定向
  __webpack_public_path__ = window.__INJECTED_PUBLIC_PATH_BY_QIANKUN__
  // 注入
  inject({
    InitVue,
    InitVueRouter,
  })
}

// 导出给基座调用
export { bootstrap, mount, unmount }

// microChildApp.js
import constantRoutes from './router'
// 用于映射到基座
import actions from './actions'

let instance = null // 接收Vue实例
let router = null // 接收VueRouter实例

let _vueFn = null // Vue实例方法
let _routerFn = null // VueRouter实例方法

// 注入变量到微前端环境
function inject({ InitVue, InitVueRouter }) {
  _vueFn = InitVue
  _routerFn = InitVueRouter
}

// 渲染
function render({ props = {}, InitVue = _vueFn, InitVueRouter = _routerFn }) {
  // 接收子项目生命周期传递的值
  if (props) actions.setActions(props)
  // 创建VueRouter实例
  router = InitVueRouter(constantRoutes)
  // 创建Vue实例
  instance = InitVue(router)
}

// 以下是子项目生命周期
async function bootstrap() {
  console.log('micro-app-search bootstraped')
}
async function mount(props = {}) {
  console.log('micro-app-search mount', props)
  render({ props })
}
async function unmount() {
  console.log('micro-app-search unmount', props)
  instance.$destroy()
  instance = null
  router = null
}

export { actions, inject, render, bootstrap, mount, unmount }

// actions.js
function emptyAction() {
  console.warn('Current execute action is empty!')
}

class Actions {
  actions = {
    onGlobalStateChange: emptyAction,
    setGlobalState: emptyAction,
  }

  setActions(actions) {
    this.actions = actions
  }

  onGlobalStateChange(...args) {
    return this.actions.onGlobalStateChange(...args)
  }

  setGlobalState(...args) {
    return this.actions.setGlobalState(...args)
  }
}

const actions = new Actions()
export default actions

  this.$store.state.base.routePath = state.routePath

# nginx 配置

# 注意点

  • 基座打印 window：Window:{...}
  • 子项目打印 window：Proxy {...}
  • 子项目获取 window：globalThis、self或者 window[key]
  • 子项目资源重定向：window.__INJECTED_PUBLIC_PATH_BY_QIANKUN__
  • 判断是否为微前端环境window.__POWERED_BY_QIANKUN__
  • 微前端环境下 Vue.js devtools 无法查询到子项目内容。
  • 基座与子项目保持低耦合，要让子项目能单独运行
  • 通过基座去启动子项目时，需要让基座引入公共资源
  • 子项目独立运行时，需要引用基座引用的那些公共资源
  • 可提供给子项目的组件放基座中，用路径修饰符去引
