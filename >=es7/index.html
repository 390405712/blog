# Stage

  • Stage 0: strawman——最初想法的提交。
  • Stage 1: proposal（提案）——由 TC39 至少一名成员倡导的正式提案文件，该文件包括 API 事例。
  • Stage 2: draft（草案）——功能规范的初始版本，该版本包含功能规范的两个实验实现。
  • Stage 3: candidate（候选）——提案规范通过审查并从厂商那里收集反馈
  • Stage 4: finished（完成）——提案准备加入 ECMAScript，但是到浏览器或者 Nodejs 中可能需要更长的时间。

# for wait of

  • 异步迭代器
function Gen(time) {
  return new Promise(function(resolve, reject) {
    setTimeout(function() {
      resolve(time)
    }, time)
  })
}
async function test() {
  let arr = [Gen(2000), Gen(100), Gen(3000)]
  for await (let item of arr) {
    console.log(Date.now(), item)
  }
}
test()
// 1575536194608 2000
// 1575536194608 100
// 1575536195608 3000

# Promise.prototype.finally()

  • 在执行 then()和 catch()后，都会执行 finally()
fetch('https://www.google.com')
  .then((response) => {
    console.log(response.status)
  })
  .catch((error) => {
    console.log(error)
  })
  .finally(() => {
    document.querySelector('#spinner').style.display = 'none'
  })

# 新正则

  • s(dotAll)flag
console.log(/foo.bar/.test('foo\nbar')) // false
console.log(/foo.bar/s.test('foo\nbar')) // true

const re = /foo.bar/s
// const re = new RegExp('foo.bar', 's')
console.log(re.test('foo\nbar')) // true
console.log(re.dotAll) // true
console.log(re.flags) // 's'
  • 命名获取组
const re = /(\d{4})-(\d{2})-(\d{2})/
const match = re.exec('2019-01-01')
console.log(match[0]) // 2019-01-01
console.log(match[1]) // 2019
console.log(match[2]) // 01
console.log(match[3]) // 01

const re = /(?<year>\d{4})-(?<month>\d{2})-(?<day>\d{2})/
const match = re.exec('2019-01-01')
console.log(match.groups) // {year: "2019", month: "01", day: "01"}
console.log(match.groups.year) // 2019
console.log(match.groups.month) // 01
console.log(match.groups.day) // 01
  • Lookbehind 后行断言
// (?<…)是后行断言的符号，(?..)是先行断言的符号，然后结合 =(等于)、!(不等)、\1(捕获匹配)。
let test1 = 'hello world'
console.log(test1.match(/hello(?=\sworld)/))
// ["hello", index: 0, input: "hello world", groups: undefined]

let test2 = 'world hello'
console.log(test2.match(/(?<=world\s)hello/))
// ["hello", index: 6, input: "world hello", groups: undefined]

# ES10

# Array.prototype.flat()

newArray = arr.flat(depth) // depth是指定要提取嵌套数组的结构深度，默认值为 1

const numbers1 = [1, 2, [3, 4, [5, 6]]]
numbers1.flat() // [1, 2, 3, 4, [5, 6]]
const numbers2 = [1, 2, [3, 4, [5, 6]]]
numbers2.flat(2) // [1, 2, 3, 4, 5, 6]

# Array.prototype.flatMap()

let arr = [1, 2, 3]
console.log(arr.map(item => [item * 2]) // [[2], [4], [6]]
console.log(arr.map(item => [item * 2]).flat()) // [2, 4, 6]
console.log(arr.flatMap(item => [item * 2])) // [2, 4, 6]

# Object.fromEntries()

const object = { x: 23, y: 24 }
const entries = Object.entries(object) // [['x', 23], ['y', 24]]
const result = Object.fromEntries(entries) // { x: 23, y: 24 }
