# TypeScript

# 介绍

  • 该语言由微软开发，是 JavaScript 的超集
  • ts 最终会转译成 js
  • npm install -g typescript

# 基础类型

// 【布尔值】
const isDone: boolean = false

// 【数字】
const count: number = 1

// 【字符串】
const str: string = '123'

// 【对象】
declare function create(o: object | null): void

create({ prop: 0 }) // OK
create(null) // OK
create(42) // Error

// 【数组】
const arr1: number[] = [1, 2, 3]
const arr2: Array<number> = [1, 2, 3]
let arr3: [string, number]
arr3 = ['str', 1] // true
arr3 = [1, 'str'] // false

// 【枚举】
enum Color1 {
  Red, // 0
  Green, // 1
  Blue, // 2
}
// { '0': 'Red', '1': 'Green', '2': 'Blue', Red: 0, Green: 1, Blue: 2 }
const c1: Color1 = Color1.Green // 1

enum Color2 {
  Red = 1, // 1
  Green, // 2
  Blue, // 3
}
// { '1': 'Red', '2': 'Green', '3': 'Blue', Red: 1, Green: 2, Blue: 3 }
let c2: Color2 = Color2.Green // 2

// 【Any】
const anyType: any = 1 || '1' || true || [] || {}

// 【void、null、undefined】
const isNoType: void = undefined || null
const isNoType: null = undefined || null
const isNoType: undefined = undefined || null

// 【Never】
// 返回never的函数必须存在无法达到的终点
function error(message: string): never {
  throw new Error(message)
}
// 返回never的函数必须存在无法达到的终点
function infiniteLoop(): never {
  while (true) {}
}

// 【断言】
const val: any = 'str'
const valLen: number = (<string>val).length
// 或者
const valLen1: number = (val as string).length

# 可选属性

interface config {
  color?: string
  width?: number
}
function createFn(config: config): { color: string; width: number } {
  let newConfig = { color: 'white', width: 111 }
  newConfig.color = config.color
  newConfig.width = config.width
  return newConfig
}
const val = createFn({ color: 'blue' }) // { color: 'blue', width: undefined }

# 混合类型

interface countIF {
  (start: number): string
  interval: number
  reset(): void
}
function getCounter(): countIF {
  let counter = <countIF>function(start: number) {}
  counter.interval = 123
  counter.reset = function() {}
  return counter
}
let c = getCounter()
// [Function: counter] { interval: 123, reset: [Function (anonymous)] }
c(10)
// [Function: counter] { interval: 123, reset: [Function (anonymous)] }
c.reset()
// [Function: counter] { interval: 123, reset: [Function (anonymous)] }
c.interval = 5.0
// [Function: counter] { interval: 5, reset: [Function (anonymous)] }

# 公共、私有、受保护修饰符

  • 默认：public
class a {
  public name: string
  public constructor(val: string) {
    this.name = val
  }
}
new a('张三').name // 张三
  • 私有：private(子类、实例无法调用)
class a {
  private name: string
  constructor(val: string) {
    this.name = val
  }
}
new a('张三').name // error

class b extends a {
  constructor() {
    super('b')
  }
}

class c {
  private name: string
  constructor(val: string) {
    this.name = val
  }
}
let newA = new a('a')
let newB = new b('b')
let newC = new c('c')

newA = newB
newA = newC // error
  • 受保护：protected(可以被子类调用，无法被实例调用)
class a {
  // 这里如果变成 private 子类用 `this.name` 会报错
  protected name: string
  constructor(val: string) {
    this.name = val
  }
}

class b extends a {
  private age: string
  constructor(name: string, age: string) {
    super(name)
    this.age = age
  }
  public getVal() {
    // 在这里可以访问
    return `${this.name} ${this.age}`
  }
}

let newB = new b('张三', 1)
console.log(newB.getVal())
console.log(newB.name) // error 无法通过实例访问

class a {
  protected name: string
  protected constructor(val: string) {
    this.name = val
  }
}
new a('z') // 类“a”的构造函数是受保护的，仅可在类声明中访问

getLength<string>('123')
