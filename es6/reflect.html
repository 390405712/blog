# Reflect

  • 第一个参数必须为对象

# defineProperty(target, prop, desc)

  • target:分别是目标对象 obj
  • prop:被编辑的属性名 prop
  • desc:以及该属性的描述 descriptor
function MyDate() {
  /*…*/
}

// 旧写法 返回值是设置的对象
Object.defineProperty(MyDate, 'now', {
  value: () => Date.now(),
})

// 新写法 返回值是true
Reflect.defineProperty(MyDate, 'now', {
  value: () => Date.now(),
})

const p = new Proxy(
  {},
  {
    defineProperty(target, prop, desc) {
      console.log(desc) // {value: "bar", writable: true, enumerable: true, configurable: true}
      return Reflect.defineProperty(target, prop, desc) // true
    },
  }
)

p.foo = 'bar'
// {value: "bar", writable: true, enumerable: true, configurable: true}
p.foo // "bar"

obj:{
  a:{
    // value:1, // 与get、set不能同时存在
    get:getter(), // 取值
    set:setter(), // 改值
    writable:true, // 可写
    enumerable:true, // 可枚举
    configurable:true // 以上配置是否可修改
  }
}
  • 数据描述符value:属性值
  • 存取描述符get:取值函数
  • 存取描述符set:改值函数
  • 基础描述符writable:value 是否可改写
  • 基础描述符enumerable：是否可枚举，为 false 时，Objcet.keys() 和 for...in 都无法获取到被定义的属性
  • 基础描述符configurable：以上配置是否可被修改，为 false 时将无法被 delete
  • 注意：数据描述符和存取描述符不能同时存在

# get(target, prop, receiver)

var target = {
  foo: 1,
  bar: 2,
  get baz() {
    return this.foo + this.bar
  },
}
var myReceiverObject = {
  foo: 4,
  bar: 4,
}
Reflect.get(target, 'foo') // 1
Reflect.get(target, 'bar') // 2
Reflect.get(target, 'baz') // 3
Reflect.get(target, 'baz', myReceiverObject) // 8 函数的this绑定receiver

# set(target, prop, value, receiver)

var target = {
  foo: 1,
  set bar(value) {
    return (this.foo = value)
  },
}

target.foo // 1
Reflect.set(target, 'foo', 2)
target.foo // 2
Reflect.set(target, 'bar', 3)
target.foo // 3
var myReceiverObject = {
  foo: 0,
}
Reflect.set(target, 'bar', 1, myReceiverObject) // 赋值函数的this绑定receiver。  myReceiverObject.foo = 1
target.foo // 3
myReceiverObject.foo // 1

# has(target, prop)

var target = {
  foo: 1,
}

// 旧写法
'foo' in target // true

// 新写法
Reflect.has(target, 'foo') // true

# deleteProperty(target, prop)

const target = { foo: 'bar' }

// 旧写法
delete target.foo // 成功删除返回true

// 新写法
Reflect.deleteProperty(target, 'foo') // 成功删除返回true

# construct(target, args)

function Greeting(prop) {
  this.name = prop
}

// new 的写法
const instance = new Greeting('张三')
//Greeting {name: "张三"}

// Reflect.construct 的写法
const instance = Reflect.construct(Greeting, ['张三'])
//Greeting {name: "张三"}

# getPrototypeOf(target)

const target = new FancyThing()

// 旧写法
Object.getPrototypeOf(target) === FancyThing.prototype

// 新写法
Reflect.getPrototypeOf(target) === FancyThing.prototype

Object.getPrototypeOf(1) // Number {[[PrimitiveValue]]: 0}
Reflect.getPrototypeOf(1) // 报错

# setPrototypeOf(target, prototype)

const target = {}

// 旧写法
Object.setPrototypeOf(target, Array.prototype) // set成功返回true

// 新写法
Reflect.setPrototypeOf(target, Array.prototype) // set成功返回true

target.length // 0

Reflect.setPrototypeOf({}, null)
// true
Reflect.setPrototypeOf(Object.freeze({}), null)
// false
Object.setPrototypeOf(1, {})
// 1
Reflect.setPrototypeOf(1, {})
// TypeError: Reflect.setPrototypeOf called on non-object
Object.setPrototypeOf(null, {})
// TypeError: Object.setPrototypeOf called on null or undefined
Reflect.setPrototypeOf(null, {})
// TypeError: Reflect.setPrototypeOf called on non-object

# apply(target, thisArg, args)

const ages = [11, 33, 12, 54, 18, 96]

// 旧写法
const youngest = Math.min.apply(Math, ages) // 11
const oldest = Math.max.apply(Math, ages) // 96
const type = Object.prototype.toString.call(youngest) // [object Number]

// 新写法
const youngest = Reflect.apply(Math.min, Math, ages) // 11
const oldest = Reflect.apply(Math.max, Math, ages) // 96
const type = Reflect.apply(Object.prototype.toString, youngest, []) // [object Number]

# getOwnPropertyDescriptor(target, prop)

var target = {}
Object.defineProperty(target, 'hidden', {
  value: true,
  enumerable: false,
})

// 旧写法
var theDescriptor = Object.getOwnPropertyDescriptor(target, 'hidden') // {hidden:true}

// 新写法
var theDescriptor = Reflect.getOwnPropertyDescriptor(target, 'hidden') // {hidden:true}

const target = {}

# preventExtensions(target) 变为不可扩展对象

var target = {}

// 旧写法
Object.preventExtensions(target) // 成功时返回Object {}

// 新写法
Reflect.preventExtensions(target) // 成功时返回true

// ES5 环境
Object.preventExtensions(1) // 报错

// ES6 环境
Object.preventExtensions(1) // 1

// 新写法
Reflect.preventExtensions(1) // 报错

# ownKeys(target)

var myObject = {
  foo: 1,
  bar: 2,
  [Symbol.for('baz')]: 3,
  [Symbol.for('bing')]: 4,
}

// 旧写法
Object.getOwnPropertyNames(myObject)
// ['foo', 'bar']

Object.getOwnPropertySymbols(myObject)
//[Symbol(baz), Symbol(bing)]

// 新写法
Reflect.ownKeys(myObject)
// ['foo', 'bar', Symbol(baz), Symbol(bing)]
